# 安装 MYSQL

sudo apt-get update
sudo apt-get install mysql-server

# 初始化环境

sudo mysql_secure_installation

# 远程访问

编辑文件

$ vim /etc/mysql/mysql.conf.d/mysqld.cnf

把下面内容注释掉

bind-address           = 127.0.0.1
