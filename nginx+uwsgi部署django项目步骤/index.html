# 2. 配置 nginx

内容：

server{
        # 指定本项目监听端口,浏览器输入端口
        listen 80;
        # 域名
        server_name 127.0.0.1:80;
        # 指定字符集
        charset utf-8;
        # 指定收集静态文件路径
        location /static{
            alias /home/sourcelist/project/Sourcelist/static;
          }
        # 指定收集媒体文件路径
        location /media{
          alias /home/sourcelist/project/Sourcelist/media;
          }
        # 和uwsgi通信端口和通信文件
        location /{
          include uwsgi_params;
          uwsgi_pass 127.0.0.1:8001;
      }            
    }
